type effect_handler =
  1. | EffectHandler of pattern * expr * continue_val
    (*

    try x with | E k -> continue k 0 - the part after with is the effect handler

    *)
and expr =
  1. | EConst of const
    (*

    1

    *)
  2. | EBinaryOperation of bin_op * expr * expr
    (*

    1 + 1

    *)
  3. | EUnaryOperation of un_op * expr
    (*

    -1, not true

    *)
  4. | EIdentifier of id
    (*

    let f = 5 - here f is identifier

    *)
  5. | EApplication of expr * expr
    (*

    f x

    *)
  6. | EFun of pattern * expr
    (*

    fun x -> x

    *)
  7. | EDeclaration of id * expr * expr option
    (*

    let f = 1

    *)
  8. | ERecDeclaration of id * expr * expr option
    (*

    let rec f x = f (x-1)

    *)
  9. | EEffectDeclaration of id * effect_types_annotation
    (*

    effect DevisionByZero : int effect

    *)
  10. | EEffectWithArguments of id * expr
    (*

    DevisionByZero 0

    *)
  11. | EEffectWithoutArguments of id
    (*

    DevisionByZero

    *)
  12. | EIfThenElse of expr * expr * expr
    (*

    if x = 1 then x else x - 1

    *)
  13. | EList of expr list
    (*

    1 ; 2 ; 3

    *)
  14. | EListCons of expr * expr
    (*

    1 :: 2 :: 3 ::

    *)
  15. | ETuple of expr list
    (*

    (1, true, 'c')

    *)
  16. | EMatchWith of expr * (pattern * expr) list
    (*

    match x with | hd :: tl -> hd | _ -> 0

    *)
  17. | ETryWith of expr * effect_handler list
    (*

    try x with | Effect1 k -> continue k 0 | Effect2 k -> continue k 1

    *)
  18. | EEffectPerform of expr
    (*

    perform DevisionByZero

    *)
  19. | EEffectContinue of continue_val * expr
    (*

    continue k 0

    *)
