type value =
  1. | VInt of int
  2. | VBool of bool
  3. | VChar of char
  4. | VString of string
  5. | VUnit
  6. | VTuple of value list
  7. | VList of value list
  8. | VFun of Ast.pattern * Ast.expr * enviroment
  9. | VRecFun of Ast.id * value
  10. | VEffectDeclaration of Ast.id
  11. | VEffectWithArguments of Ast.id * value
  12. | VEffectWithoutArguments of Ast.id
  13. | VEffectContinue of Ast.continue_val
  14. | VThrowingValue of value
and enviroment = (string, value, Base.String.comparator_witness) Base.Map.t
and handlers = (string, Ast.pattern * Ast.expr * Ast.continue_val, Base.String.comparator_witness) Base.Map.t
