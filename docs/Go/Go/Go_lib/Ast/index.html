type type' =
  1. | StringType
  2. | IntegerType
  3. | BooleanType
  4. | FuncType of signature

  1. args : arg list;

  2. | Void

type binary_operation =
  1. | MinusBin
  2. | Asterisk
  3. | Plus

  5. | Equal

  7. | Less
  8. | LessEqual
  9. | Greater
  10. | GreaterEqual
  11. | And
  12. | Or

type expression =
  1. | Const of identifier
  2. | Identifier of string
  3. | BinaryOp of expression * binary_operation * expression
  4. | UnaryOp of unaryop * expression
  5. | FunCall of expression * expression list
  6. | AnonFunc of signature * block
  7. | PrintCall of expression list
  8. | LenCall of expression
and identifier =
  1. | Integer of int
  2. | String of string
  3. | Bool of bool
  4. | Nil
and var_declaration = string * type' * expression
and func_declaration = string * signature * block
and global_declaration =
  1. | VarTopLevelDeclaration of var_declaration
  2. | FuncTopLevelDeclaration of func_declaration
and block = statements list
and else_ = block
and statements =
  1. | IfElseStatement of expression * block * else_
  2. | ForStatement of expression * block
  3. | ReturnStatement of expression option
  4. | BlockStatement of block
  5. | ExpressinStatement of expression
  6. | AssignStatement of expression * expression
  7. | VarStatement of var_declaration
  8. | BreakStatement
  9. | ContinueStatement
