  1. | TInt
    (*

    int type

  2. | TBool
    (*

    bools type

  3. | TArrow of typ * typ
    (*

    type -> type

    *)
  4. | TVar of int
    (*

    var type

    *)
  5. | TList of typ

  1. | Occurs_check
  2. | No_variable of string
