type value =
  1. | VString of string
    (*

    "string"

    *)
  2. | VBool of bool
    (*

    true

    *)
  3. | VInt of int
    (*

    42

    *)
  4. | VList of value list
    (*

    1;2;3

    *)
  5. | VTuple of value list
    (*

    (1,2,3)

    *)
  6. | VFun of Ast.pattern * Ast.expr * (Ast.name * value) list
  7. | VLetWAPat of Ast.name * value
    (*

    Let without active patterns, let x y = y * 5

    *)
  8. | VLetAPat of Ast.name list * value
    (*

    Let with active patterns, let (|Even|Odd|) value = .., let (|Even|_|) value = .., let (|Even|) value = ..

    *)
  9. | VCases of Ast.name
    (*

    Even, Odd, ...

    *)
  10. | VNone
