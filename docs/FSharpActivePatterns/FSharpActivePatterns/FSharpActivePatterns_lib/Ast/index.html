type pattern =
  1. | Wild
    (*

    _

    *)
  2. | Const of const
    (*

    string, bool or int

    *)
  3. | Var of name
    (*

    variable

    *)
  4. | Tuple of pattern list
    (*

    (a, b)

    *)
  5. | PCon of pattern * pattern
    (*

    hd::tl

    *)
  6. | Case of name * pattern list
    (*

    in patmatch

    *)
val equal_pattern : pattern -> pattern -> Ppx_deriving_runtime.bool
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | ConstExpr of const
  2. | VarExpr of name
    (*

    x = 5

    *)
  3. | ListExpr of expr * expr
    (*

    1;2;3

    *)
  4. | TupleExpr of expr list
    (*

    (1,2,3)

    *)
  5. | BinExpr of binary_op * expr * expr
    (*

    1 + 5 - 3

    *)
  6. | IfExpr of expr * expr * expr
    (*

    if a then b else c

    *)
  7. | LetInExpr of bool * name * expr * expr
    (*

    let plusfive x = let five a = a + 5 in five x

    *)
  8. | AppExpr of expr * expr
    (*

    sq 5

    *)
  9. | FunExpr of pattern * expr
    (*

    fun x -> x * x

    *)
  10. | MatchExpr of expr * (pattern * expr) list
    (*

    match input with | 2 -> 5 | 5 -> 10

    *)
  11. | CaseExpr of name
    (*

    Even, Odd

    *)
val equal_expr : expr -> expr -> Ppx_deriving_runtime.bool
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type struct_inter =
  1. | Let of bool * name * expr
  2. | LetAct of name list * expr
  3. | Expression of expr
val equal_struct_inter : struct_inter -> struct_inter -> Ppx_deriving_runtime.bool
val pp_struct_inter : Ppx_deriving_runtime.Format.formatter -> struct_inter -> Ppx_deriving_runtime.unit
val show_struct_inter : struct_inter -> Ppx_deriving_runtime.string
