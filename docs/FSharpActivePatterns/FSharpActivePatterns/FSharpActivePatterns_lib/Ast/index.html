type pattern =
  1. | Wild
    (*

    _

    *)
  2. | Const of const
    (*

    string, bool or int

    *)
  3. | Var of name
    (*

    variable

    *)
  4. | Tuple of pattern list
    (*

    (a, b)

    *)
  5. | List of pattern list
    (*

    1;2;3

    *)
  6. | Case of name * pattern list
    (*

    in patmatch

    *)

type expr =
  1. | ConstExpr of const
  2. | VarExpr of name
    (*

    x = 5

    *)
  3. | ListExpr of expr list
    (*

    1;2;3

    *)
  4. | TupleExpr of expr list
    (*

    (1,2,3)

    *)
  5. | BinExpr of binary_op * expr * expr
    (*

    1 + 5 - 3

    *)
  6. | IfExpr of expr * expr * expr
    (*

    if a then b else c

    *)
  7. | LetExpr of bool * name * expr
    (*

    let sq x = x * x, bool - is rec or not

    *)
  8. | LetActExpr of name list * expr
    (*

    let (|Even|Odd|) value = if value % 2 = 0 then Even else Odd

    *)
  9. | AppExpr of expr * expr
    (*

    sq 5

    *)
  10. | FunExpr of pattern * expr
    (*

    fun x -> x * x

    *)
  11. | MatchExpr of expr * (pattern * expr) list
    (*

    match input with | 2 -> 5 | 5 -> 10

    *)
  12. | CaseExpr of name
    (*

    (|Even|Odd|)

    *)
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
