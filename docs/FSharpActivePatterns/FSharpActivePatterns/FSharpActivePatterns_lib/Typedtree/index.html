type ac = id * typ list
and typ =
  1. | Prim of string
    (*

    Available ground types

    *)
  2. | Ty_var of binder
    (*

    'a, 'b types

    *)
  3. | Arrow of typ * typ
    (*

    Function type: 'a -> 'a

    *)
  4. | List of typ
    (*

    List type: int list

    *)
  5. | Tuple of typ list
    (*

    Typle type: int, string means (int, string)

    *)
  6. | ActiveCases of binder * ac list
