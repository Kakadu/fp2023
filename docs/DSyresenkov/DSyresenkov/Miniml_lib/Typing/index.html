type id = string
val pp_id : Ppx_deriving_runtime.Format.formatter -> id -> Ppx_deriving_runtime.unit
val show_id : id -> Ppx_deriving_runtime.string
module VarSet : Stdlib.Set.S with type elt = int
type base_type =
  1. | BInt
    (*

    Basic integer type

    *)
  2. | BBool
    (*

    Basic bool type

    *)
  3. | BUnit
    (*

    Unit type

    *)
val pp_base_type : Ppx_deriving_runtime.Format.formatter -> base_type -> Ppx_deriving_runtime.unit
val show_base_type : base_type -> Ppx_deriving_runtime.string
type ty =
  1. | TBase of base_type
    (*

    Type of integer

    *)
  2. | TVar of int
  3. | TArrow of ty * ty
    (*

    Type of function ty1 -> ty2

    *)
  4. | TTuple of ty * ty * ty list
    (*

    Type of tuple

    *)
  5. | TList of ty
    (*

    Type of list

    *)
val pp_ty : Ppx_deriving_runtime.Format.formatter -> ty -> Ppx_deriving_runtime.unit
val show_ty : ty -> Ppx_deriving_runtime.string
type error =
  1. | OccursCheckFailed of int * ty
    (*

    OCaml's Occurs check

    *)
  2. | NoVariable of id
    (*

    Attempt to use non-initialized variable

    *)
  3. | UnificationFailed of ty * ty
    (*

    Failed to unify left and right types

    *)
  4. | PatternRebound
    (*

    Attempt to use in pattern single bound two times (h :: h :: tl)

    *)
  5. | NotImplemented
    (*

    Still not implemented features

    *)
val pp_error : Ppx_deriving_runtime.Format.formatter -> error -> Ppx_deriving_runtime.unit
val show_error : error -> Ppx_deriving_runtime.string
val pp_type : Stdlib.Format.formatter -> ty -> unit
val print_ty : ty -> unit
type scheme =
  1. | S of VarSet.t * ty
