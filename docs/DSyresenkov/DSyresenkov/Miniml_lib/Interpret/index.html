type value =
  1. | VInt of int
  2. | VBool of bool
  3. | VUnit
  4. | VTuple of value * value * value list
  5. | VList of value list
  6. | VFun of Ast.id * Ast.expr * bindings
and bindings = (Ast.id, value, Base.String.comparator_witness) Base.Map.t
val pp_value : Stdlib.Format.formatter -> value -> unit
type interpret_result = {
  1. id : Ast.id option;
  2. value : value;
  3. ty : Typing.ty;
}
val pp_interpret_result : Stdlib.Format.formatter -> interpret_result -> unit
type error =
  1. | DivisionByZero
  2. | UnboundValue of Ast.id
  3. | IncorrectType of value
  4. | NotAFunction of value
  5. | LetWithoutIn
    (*

    Inner let expressions without in are forbidden

    *)
  6. | TypeInferFailed of Typing.error
  7. | NoMatchCase of value
  8. | NotImplemented
val pp_error : Stdlib.Format.formatter -> error -> unit
val interpret : ?tyenv:Infer.TypeEnv.t -> Ast.program -> (bindings * interpret_result list, error) Result.t
