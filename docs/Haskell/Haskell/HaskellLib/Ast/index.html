type binding = pat * expr

Type representing binding from patterns to expressions.

and alt = pat * expr

Type representing case alternative in pattern matching.

and expr =
  1. | ExprLit of lit
    (*

    Literal value 1, 'a', "abc"

    *)
  2. | ExprVar of id
    (*

    Variable reference x

    *)
  3. | ExprFunc of binding
    (*

    Function/closure definition \x -> x + 1

    *)
  4. | ExprApp of expr * expr
    (*

    Function application f x

    *)
  5. | ExprIf of expr * expr * expr
    (*

    If expression if x then y else z

    *)
  6. | ExprTuple of expr list
    (*

    Tuple (x, y)

    *)
  7. | ExprCons of expr * expr
    (*

    List construction x:xs

    *)
  8. | ExprNil
    (*

    Empty list []

    *)
  9. | ExprCase of expr * alt list
    (*

    Case expression case (m,ys) of (0,_) -> [] (_,[]) -> [] (n,x:xs) -> x : take (n-1) xs

    *)
  10. | ExprBinOp of bin_op * expr * expr
    (*

    Binary operation a + b

    *)
  11. | ExprUnOp of un_op * expr
    (*

    Unary operation not a

    *)
  12. | ExprLet of binding list * expr
    (*

    Local binding let x = 1 in x + 2

    *)

Type representing possible expressions

and pat =
  1. | PatWild
    (*

    Wildcard pattern _

    *)
  2. | PatVar of id
    (*

    Variable pattern x

    *)
  3. | PatLit of lit
    (*

    Literal pattern 1, 'a', "abc"

    *)
  4. | PatTuple of pat list
    (*

    Tuple pattern (x, y)

    *)
  5. | PatCons of pat * pat
    (*

    List constructor pattern x:xs

    *)
  6. | PatNil
    (*

    Empty list pattern []

    *)

Type representing possible patterns in pattern-matching.

and lit =
  1. | LitInt of int
    (*

    Integer literal 7

    *)
  2. | LitBool of bool
    (*

    Boolean literal True

    *)
  3. | LitChar of char
    (*

    Character literal 'a'

    *)
  4. | LitString of string
    (*

    String literal "abc"

    *)

Type representing literals

and decl =
  1. | DeclLet of binding
    (*

    Let binding declaration x = 5

    *)

Type representing a single declaration.

and prog = decl list

Type representing a whole program as a series of declarations.
