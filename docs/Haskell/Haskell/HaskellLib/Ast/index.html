type binding = pat * expr
and alt = pat * expr

Alternatives in case expression

and expr =
  1. | ExprLit of lit
  2. | ExprVar of id
  3. | ExprFunc of binding
    (*

    Essentially a closure

    *)
  4. | ExprApp of expr * expr
  5. | ExprIf of expr * expr * expr
  6. | ExprTuple of expr list
  7. | ExprCons of expr * expr
  8. | ExprNil
  9. | ExprCase of expr * alt list
  10. | ExprBinOp of bin_op * expr * expr
  11. | ExprUnOp of un_op * expr
  12. | ExprLet of binding list * expr
    (*

    Local binding let x = 2; y = 3 in e x y

    *)
and pat =
  1. | PatWild
    (*

    Wildcard Pat _

    *)
  2. | PatVar of id
    (*

    Variable Pat (name binding) x

    *)
  3. | PatLit of lit
    (*

    Literal pat 1, 'a', "hello"

    *)
  4. | PatTuple of pat list
    (*

    Tuple pat (x, y)

    *)
  5. | PatCon of pat * pat
    (*

    Constructor pat

    *)
  6. | PatAs of id * pat
    (*

    As-pat list@(x:xs)

    *)
  7. | PatList of pat list
  8. | PatRec of id * pat list
    (*

    Record pat Point {x = 3, y}

    *)
  9. | PatNil
    (*

    *)
and lit =
  1. | LitInt of int
  2. | LitChar of char
  3. | LitString of string
  4. | LitFloat of float
and decl =
  1. | DeclLet of binding
and prog = decl list
