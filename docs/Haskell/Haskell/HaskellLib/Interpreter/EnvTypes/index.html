Module Interpreter.EnvTypes

type res = (value, err) LazyResult.t
and environment = (string, res) Stdlib.Hashtbl.t
and value =
  1. | ValInt of int
  2. | ValBool of bool
  3. | ValString of string
  4. | ValChar of char
  5. | ValNil
  6. | ValList of res * res
  7. | ValTuple of res list
  8. | ValFun of Ast.pat * Ast.expr * environment
and err =
  1. | NotInScopeError of string
  2. | DivisionByZeroError
  3. | NonExhaustivePatterns of string
  4. | TypeMismatch
