type expr =
  1. | Bin_op of bin_op * expr * expr
  2. | Un_op of un_op * expr
  3. | Var_Declaration of types * name
    (*

    Variable declaration int a

    *)
  4. | Access_By_Point of expr * expr
    (*

    Access by point to class member a.b.c

    *)
  5. | Exp_Assign of expr * expr
    (*

    Declare and assign a value to a variable int a = 1

    *)
  6. | Exp_Val of value
  7. | Exp_Name of name
  8. | Exp_Args of expr list
  9. | Method_invoke of expr * expr
    (*

    Calling a method Foo.a(i, 9)

    *)
  10. | Exp_Body of expr list
    (*

    Method body

    *)
  11. | Exp_Class of class_modifier option * name * member list option
  12. | If_Else of expr * expr * expr option
  13. | Return of expr option
and field =
  1. | Field of types * name * expr option
and methods =

  2. | Method of method_modifier list option * types * name * expr * expr
