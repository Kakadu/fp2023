type name = string
val pp_name : Stdlib.Format.formatter -> name -> unit
val show_name : name -> string
type num = int
val pp_num : Stdlib.Format.formatter -> num -> unit
val show_num : num -> string
type oper = string
val pp_oper : Stdlib.Format.formatter -> oper -> unit
val show_oper : oper -> string
type var = string
val pp_var : Stdlib.Format.formatter -> var -> unit
val show_var : var -> string
type atom =
  1. | Name of name
  2. | Oper of oper
val pp_atom : Stdlib.Format.formatter -> atom -> unit
val show_atom : atom -> string
type const =
  1. | Num of num
  2. | Atom of atom
val pp_const : Stdlib.Format.formatter -> const -> unit
val show_const : const -> string
type term =
  1. | Const of const
  2. | Var of var
  3. | Relation of {
    1. atom : atom;
    2. terms : term list;
    }
val pp_term : Stdlib.Format.formatter -> term -> unit
val show_term : term -> string
type many_term =
  1. | Many_term of term list
val pp_many_term : Stdlib.Format.formatter -> many_term -> unit
val show_many_term : many_term -> string
