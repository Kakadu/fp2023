  5. | Null

type expression =
  1. | UnOp of un_op * expression
  2. | BinOp of bin_op * expression * expression
  3. | Const of typename
  4. | Var of string
  5. | FunctionCall of expression * expression list
  6. | AnonFunction of string list * statement
  7. | ArrowFunction of string list * statement
  8. | ArrayList of expression list
  9. | ObjectDef of (expression * expression) list
and var_init = {
  1. var_identifier : string;
  2. is_const : bool;
  3. value : expression;
}
and fun_init = {
  1. fun_identifier : string;
  2. arguments : string list;
  3. body : statement;
}
and loop = {
  1. loop_init : statement option;
  2. loop_condition : expression;
  3. loop_change : expression option;
  4. loop_body : statement;
}
and statement =
  1. | Block of statement list
  2. | Expression of expression
  3. | VarInit of var_init
  4. | FunInit of fun_init
  5. | If of expression * statement * statement
  6. | Loop of loop
  7. | Return of expression
  8. | Programm of statement list
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
val pp_var_init : Ppx_deriving_runtime.Format.formatter -> var_init -> Ppx_deriving_runtime.unit
val show_var_init : var_init -> Ppx_deriving_runtime.string
val pp_fun_init : Ppx_deriving_runtime.Format.formatter -> fun_init -> Ppx_deriving_runtime.unit
val show_fun_init : fun_init -> Ppx_deriving_runtime.string
val pp_loop : Ppx_deriving_runtime.Format.formatter -> loop -> Ppx_deriving_runtime.unit
val show_loop : loop -> Ppx_deriving_runtime.string
val pp_statement : Ppx_deriving_runtime.Format.formatter -> statement -> Ppx_deriving_runtime.unit
val show_statement : statement -> Ppx_deriving_runtime.string
