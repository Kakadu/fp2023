type var_init = {
  1. var_identifier : string;
  2. is_const : bool;
  3. var_type : typename;
  4. value : expression option;
}

  1. fun_identifier : string;
  2. arguments : expression list;
  3. body : statement option;
}

  3. | VarDeck of var_init
