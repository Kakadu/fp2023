type types =
  1. | FInt of int
    (*

    integer number: ..., 0, 1, 2, ...

    *)
  2. | FString of string
    (*

    string values: "Ocaml"

    *)
  3. | FBool of bool
    (*

    boolean values: true and false

    *)
  4. | FNil
    (*

    *)
  5. | FUnit
    (*

    ()

    *)
  6. | FFloat of float
    (*

    float number: ..., 0.1, ..., 1.2, ...

    *)
  7. | Measure_float of types * measure_type
    (*

    5.0<cm>

    *)
and measure_type =
  1. | SMeasure of string * pow
    (*

    single measure: <m>

    *)
  2. | MMeasure of measure_type * binary_op * measure_type
    (*

    multiple measure: <m / sec * h ... >

    *)
and pow =
  1. | Pow of types
    (*

    ^

    *)
val equal_types : types -> types -> Ppx_deriving_runtime.bool

    _

    (*

    tuple pattern: (z, v)

    *)
  5. | PList of pattern list
    (*

    list pattern 1; 2

    *)
  6. | PCons of pattern * pattern
    (*

    hd::tl pattern

    *)

type expression =
  1. | EConst of types
    (*

    constant

    *)
  2. | EVar of id
    (*

    variable

    *)
  3. | EBinaryOp of binary_op
    (*

    binary operation

    *)
  4. | EList of expression list
    (*

    list

    *)
  5. | ETuple of expression list
    (*

    tuple

    *)
  6. | EApp of expression * expression
    (*

    application

    *)
  7. | EIfElse of expression * expression * expression
    (*

    if z then v else n

    *)
  8. | ELet of string * id * expression
    (*

    let z = ... or let rec z = ...

    *)
  9. | EFun of pattern * expression
    (*

    fun z -> z + z

    *)
  10. | EMatch of expression * (pattern * expression) list
    (*

    match

    *)
  11. | EMeasure of measure_init
    (*

    measure

    *)
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
