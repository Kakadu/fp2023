type expression =
  1. | Exp_constant of const
    (*

    Expressions constant such as 1, true

    *)
  2. | Exp_unary_op of unary_op * expression
  3. | Exp_bin_op of bin_op * expression * expression
  4. | Exp_ident of ident
    (*

    Identifiers such as x

    *)
  5. | Exp_tuple of expression list
    (*

    Expressions (E1, ..., En) Invariant: n >= 2

    *)
  6. | Exp_function of pattern * expression
    (*

    fun P1 -> E

    *)
  7. | Exp_let of decl * expression
    (*

    Exp_let({d_rec=flag; P; E}, E') represents:

    • let P = E in E' when flag is Nonrecursive
    • let rec P = E in E' when d_rec is Recursive
    *)
  8. | Exp_match of expression * (pattern * expression) list
    (*

    match E0 with P1 -> E1 | .. | Pn -> En

    *)
  9. | Exp_ifthenelse of expression * expression * expression
    (*

    if E1 then E2 else E3

    *)
  10. | Exp_apply of expression * expression
    (*

    E0 E1

    *)
  11. | Exp_object of obj
    (*

    object ... end

    *)
  12. | Exp_send of expression * ident
    (*

    E # m

    *)
  13. | Exp_override of (ident * expression) list
    (*

    {< x1 = E1; ...; xn = En >}

    *)
  14. | Exp_list of expression * expression
    (*

    The expression such as E1::E2 This also represents lists E1; ... En via E

    *)
and decl = {
  1. d_rec : rec_flag;
  2. d_pat : pattern;
  3. d_expr : expression;
}

Represents:

  • let P = E when d_rec is Nonrecursive
  • let rec P = E when d_rec is Recursive
and obj = {
  1. o_self : pattern;
  2. o_fields : obj_field list;
}

Values of type obj represents:

  • object(selfpat) ... end
  • object ... end when o_self is Pat_any
and obj_field =
  1. | Obj_val of ident * expression
    (*

    val x = E

    *)
  2. | Obj_method of private_flag * ident * expression
    (*

    method x = E

    *)
