type parse_error =
  1. | Syntax_error of string
type object_field =
  1. | Method
  2. | Variable
type infer_error =
  1. | Occurs_check of int * Typedtree.ty
    (*

    This error can lead to non-determinism of the type inference Example: fun x -> x x

    *)
  2. | Unbound_variable of string
    (*

    Variable is not defined in local context Example: let increase = x + 1

    *)
  3. | Unification_failed of Typedtree.ty * Typedtree.ty
    (*

    Unification error between two types

    *)
  4. | Several_bounds of string
    (*

    Example: let (x, x) = 1, 3

    *)
  5. | Unreachable
    (*

    For unreachable cases like missing cases in pattern-matching, because parser doesn't recognize this construction

    *)
  6. | No_variable_rec
    (*

    Only variables are allowed as left-side of 'let rec' Example: let rec (a, b) = 5

    *)
  7. | Multiple_definition of object_field * string
    (*

    An instance variable / method must have a single definition Example: object val n = 0 val n = 5 end

    *)
  8. | Undefined_method of Typedtree.ty * string
    (*

    Occurs when sending a message to undefined method in object

    *)
  9. | Not_object of Typedtree.ty
    (*

    Occurs when trying to send a message to a non-object

    *)
  10. | Cannot_match_self
    (*

    A self-reference can only be a `self' variable

    *)
type interpreter_error =
  1. | Division_by_zero
    (*

    Example: 5 / 0

    *)
  2. | Match_failure
    (*

    May occurs with non-exhaustive pattern-matching, if value doesn't match any case

    *)
  3. | Invalid_compare_arg of string
    (*

    Occurs when trying to compare two anonymous functions

    *)
  4. | Ill_right_side_rec of string
    (*

    Represents limitations of `let rec`

    *)
  5. | Ill_typed
  6. | Unbound_var of string
type error =
  1. | Parser of parse_error
  2. | Infer of infer_error
  3. | Interpreter of interpreter_error
