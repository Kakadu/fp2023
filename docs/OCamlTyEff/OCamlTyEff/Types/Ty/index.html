type t =
  1. | Ty_tuple of t Base.list
    (*

    Ty_tuple([T1 ; ... ; Tn]) represents T1 * ... * Tn. Invariant: n >= 2.

    *)
  2. | Ty_con of Misc.Ident.t * t Base.list
    (*

    Ty_con(ident, l) represents:

    • tconstr when l=[],
    • T tconstr when l=[T],
    • (T1, ..., Tn) tconstr when l=[T1 ; ... ; Tn].
    *)
