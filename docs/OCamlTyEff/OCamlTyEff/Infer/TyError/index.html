type t =
  1. | UnificationMismatch
    (*

    Lists of types to unify have different lengths

    *)
  2. | UnificationFailTy of Types.Ty.t * Types.Ty.t
    (*

    Failed to unify two types

    *)
  3. | UnificationFailEff of Types.Eff.t * Types.Eff.t
    (*

    Failed to unify two effects

    *)
  4. | UnboundVariable of Misc.Ident.t
    (*

    Failed to find a variable in the environment

    *)
  5. | OccursInTy of Types.Var.t * Types.Ty.t
    (*

    Type variable occurs in a type

    *)
  6. | OccursInEff of Types.Var.t * Types.Eff.t
    (*

    Effect variable occurs in an effect

    *)
  7. | RecursiveEffRows
    (*

    Recursive effect rows

    *)
  8. | PatVarBoundSeveralTimes of Misc.Ident.t
    (*

    Pattern(s) bound the same variable several times. E.g. `let x, x = ..`

    *)
  9. | VarsMismatchOrPattern of Misc.Ident.t
    (*

    Sides of | pattern bound different variables. E.g. `Some x | Some y`

    *)
  10. | ConstructorArityMismatch of Misc.Ident.t
    (*

    Constructor is not applied (`Some`), or applied when not needed (`None 1`)

    *)
  11. | NotVarLHSRec
    (*

    The left hand side of the recursive binding is not a var. E.g. `let rec _ = ..`

    *)
  12. | NotAllowedRHSRec of Ast.expression
    (*

    The expression is not allowed as right-hand side of `let rec'. E.g. `let rec x = x + 1`

    *)
  13. | UnboundTypeVariable of Base.string
    (*

    Type variable is unbound in type declaration. E.g. `type bar = Bar of 'a` ('a is unbound)

    *)
  14. | UnboundType of Misc.Ident.t
    (*

    Type is unbound in type declaration. E.g. `type bar = Bar of foo` (foo is unbound)

    *)
  15. | TypeArityMismatch of Misc.Ident.t
    (*

    Type expects more/less arguments than applied. E.g. `type foo = Foo` (foo expects 0 arguments) `type bar = Bar of int foo`

    *)
  16. | NotImplemented of Base.string
    (*

    Feature is not yet implemented in the type checker. The argument contains the feature description

    *)
val pp : Base.Format.formatter -> t -> Base.unit
