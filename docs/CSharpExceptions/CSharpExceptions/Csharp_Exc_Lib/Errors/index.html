type type_check_error =
  1. | Double_definition of Ast.ident
  2. | Not_find_ident of Ast.ident
  3. | Type_mismatch
  4. | Access of string
    (*

    Errors related to access (for example, accessing a private field)

    *)
  5. | Other of string
val pp_type_check_error : Stdlib.Format.formatter -> type_check_error -> unit
val show_type_check_error : type_check_error -> string
type interpreter_error =
  1. | Runtime_error of string
  2. | Type_mismatch
  3. | Null_reference
  4. | Uninitialized_variable
  5. | Syntax_error of string
  6. | Division_by_zero
  7. | System_error of string
    (*

    Specific internal interpreter errors

    *)
  8. | User_exception of Common_types.code_ident
    (*

    If the interpreter's return result is a thrown exception

    *)
val pp_interpreter_error : Stdlib.Format.formatter -> interpreter_error -> unit
val show_interpreter_error : interpreter_error -> string
type error =
  1. | Interpret_error of interpreter_error
  2. | Type_check_error of type_check_error
val pp_error : Stdlib.Format.formatter -> error -> unit
val show_error : error -> string
