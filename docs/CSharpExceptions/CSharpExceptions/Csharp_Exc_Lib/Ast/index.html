type expr =
  1. | EConst of value_
    (*

    assignable values

    *)
  2. | EIdentifier of ident
    (*

    id of something e.g. class name; var name; method name

    *)
  3. | EMethod_invoke of expr * args
    (*

    method(a, b, c) or Class.method(a, b, c)

    *)
  4. | EPoint_access of expr * expr
    (*

    access by point e.g. A.run()

    *)
  5. | EBin_op of bin_op * expr * expr
  6. | EUn_op of un_op * expr
