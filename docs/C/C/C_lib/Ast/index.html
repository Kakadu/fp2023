type name = string
val pp_name : Ppx_deriving_runtime.Format.formatter -> name -> Ppx_deriving_runtime.unit
val show_name : name -> Ppx_deriving_runtime.string
type types =
  1. | ID_bool
  2. | ID_int32
  3. | ID_int16
  4. | ID_int8
  5. | ID_uint32
  6. | ID_uint16
  7. | ID_uint8
  8. | ID_char
  9. | ID_void
  10. | ID_float
  11. | Pointer of types
  12. | Array of int option * types
val pp_types : Ppx_deriving_runtime.Format.formatter -> types -> Ppx_deriving_runtime.unit
val show_types : types -> Ppx_deriving_runtime.string
type arg =
  1. | Arg of types * name
val pp_arg : Ppx_deriving_runtime.Format.formatter -> arg -> Ppx_deriving_runtime.unit
val show_arg : arg -> Ppx_deriving_runtime.string
type bin_op =
  1. | Add
  2. | Sub
  3. | Mul
  4. | Div
  5. | Mod
  6. | Equal
  7. | NotEqual
  8. | Less
  9. | LessOrEqual
  10. | Grow
  11. | GrowOrEqual
  12. | Or
  13. | And
  14. | Lshift
  15. | Rshift
val pp_bin_op : Ppx_deriving_runtime.Format.formatter -> bin_op -> Ppx_deriving_runtime.unit
val show_bin_op : bin_op -> Ppx_deriving_runtime.string
type un_op =
  1. | Not
    (*

    !a

    *)
  2. | Plus
    (*

    +a

    *)
  3. | Minus
    (*

    -(--a)

    *)
  4. | Address
    (*

    &a

    *)
  5. | Dereference
    (*

    *a

    *)
  6. | Pref_increment
    (*

    ++a

    *)
  7. | Pref_decrement
    (*

    --a

    *)
val pp_un_op : Ppx_deriving_runtime.Format.formatter -> un_op -> Ppx_deriving_runtime.unit
val show_un_op : un_op -> Ppx_deriving_runtime.string
type values =
  1. | V_int of int
  2. | V_char of char
  3. | V_float of float
  4. | V_void
  5. | V_null
val pp_values : Ppx_deriving_runtime.Format.formatter -> values -> Ppx_deriving_runtime.unit
val show_values : values -> Ppx_deriving_runtime.string
type expr =
  1. | Unary_expr of un_op * expr
  2. | Bin_expr of bin_op * expr * expr
  3. | Const of values
  4. | Type of types
    (*

    sizeof(int)

    *)
  5. | Func_call of name * expr list
    (*

    factorial(1, 2 + 3, 5 > 7)

    *)
  6. | Var_name of name
    (*

    return <var_name>

    *)
  7. | Cast of types * expr
    (*

    (int) a

    *)
  8. | Index of expr * expr
    (*

    a12 => Index(Index(Var_name(a), 1), 2)

    *)
  9. | Array_value of expr list
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type statement =
  1. | Var_decl of types * name * statement option
    (*

    int **b12;

    *)
  2. | Assign of expr * statement
    (*

    int n = b = 4

    *)
  3. | Expression of expr
  4. | Return of expr
  5. | Compound of statement list
    (*

    n = 4; { n = 3;... here n = 4

    }

    *)
  6. | While of expr * statement
  7. | For of statement option * expr option * expr option * statement
    (*

    for (init?; cond?; upd?) expr list

    *)
  8. | If_else of expr * statement * statement option
  9. | Break
  10. | Continue
val pp_statement : Ppx_deriving_runtime.Format.formatter -> statement -> Ppx_deriving_runtime.unit
val show_statement : statement -> Ppx_deriving_runtime.string
type func_decl =
  1. | Func_decl of types * name * arg list * statement
val pp_func_decl : Ppx_deriving_runtime.Format.formatter -> func_decl -> Ppx_deriving_runtime.unit
val show_func_decl : func_decl -> Ppx_deriving_runtime.string
type program = func_decl list
val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
