type expr =
  1. | EConst of const
    (*

    lenguage constant

    *)
  2. | EVar of name
    (*

    variable

    *)
  3. | EApply of expr * expr
    (*

    apply foo a b is EApply(EApply(EVar "foo", EVar "a"), EVar "b")

    *)
  4. | EBinOp of {
    1. op : bin_op;
    2. left : expr;
    3. right : expr;
    }
    (*

    binary operator apply op - binary operator left - first argument right - second argument

    *)
  5. | EUnOp of {
    1. op : un_op;
    2. arg : expr;
    }
    (*

    unary operator apply op - unary operator arg - arg

    *)
  6. | EList of expr list
    (*

    list constant e1; e2; e3 - list of e1, e2, e3 elements - blank list

    *)
  7. | ETuple of expr * expr * expr list
    (*

    n-tuple constant where n >= 2 always

    2-typle - ETuple(e1, e2, ) 4-typle - ETuple(e1, e2, e3; e4)

    1-tuple - never

    *)
  8. | Eite of {
    1. cond : expr;
    2. th : expr;
    3. el : expr;
    }
    (*

    if-then-else expression cond - condition th - result of ite if cond el - result of ite if not cond

    *)
  9. | EFun of {
    1. args : pattern * pattern list;
    2. expr : expr;
    }
    (*

    fun a b c -> ... expression always has at least 1 argument

    patterns - names of arguments expr - result of function

    *)
  10. | EPattern of {
    1. match_expr : expr;
    2. matches : (pattern * expr) * (pattern * expr) list;
    }
    (*

    pattern-matching match_expr - matching expression matches - list of patterns ot match

    *)
  11. | ELetIn of binding
    (*

    let and let-in statements name - name of expression/expressions rec_flag - recursive or not value - binded expr expr - where it binded

    functions represented as EFun expr

    *)
