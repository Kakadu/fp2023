type gr_ty =
  1. | GrInt
    (*

    int type

    *)
  2. | GrBool
    (*

    bool type

    *)
  3. | GrUnit
    (*

    unit type

    *)

common types

type ty =
  1. | TyGround of gr_ty
    (*

    common types

    *)
  2. | TyArrow of ty * ty
    (*

    function types

    *)
  3. | TyTuple of ty * ty * ty list
    (*

    tuple types

    *)
  4. | TyList of ty
    (*

    'ty list

    *)
  5. | TyVar of id
    (*

    type variable

    *)
